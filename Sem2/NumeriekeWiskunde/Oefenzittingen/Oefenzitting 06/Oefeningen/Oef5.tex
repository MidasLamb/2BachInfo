  \subsection{Oefening 5:}

  $
    y_{n-1}(x) + (f_{n} - y{n-1}(x_{n})) l_{n}(x) =? y_{n}(x)
  $
  \\ heeft graad $\leq$ n
  \\ Interpolerend?
  \begin{itemize}
    \item i $\neq$ \\
      $
        y_{n-1}(x_{i}) + (f_{n} - y_{n-1}(x_{n}))l_{n}(x_{i})
        \\
        \Downarrow l_{n}(x_{i}) = 0
        \\
        = y_{n-1}(x_{i}) = f_{i}
      $
    \item $i = n$ \\
      $
        y_{n-1}(x_{i}) + (f_{n} - y_{n-1}(x_{n}))l_{n}(x_{n})
        \\
        \Downarrow l_{n}(x_{n}) = 1
        \\
        = y_{n-1}(x_{n}) + f_{n} - y_{n-1}(x_{n}) = f_{n}
      $
