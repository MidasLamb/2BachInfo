        $
          \{ t.A, t.B, t.C, s.D, s.E | R(t) \wedge S(s) \wedge t.C = s.C \}
        $
      \item
        $
          \{ t | R(t) \vee S(t)  \}
        $
      \item
        $
          \{ t | R(t) \wedge S(t) \}
        $
      \item
        $
          \{ t | R(t) \wedge \neg S(t) \}
        $
      \item

        $

          \{ t.B | R(t) \wedge \forall(s) (\neq S(s) \vee ( \exists (x) R(x) \wedge t.B = x.B \wedge x.A = s.A)) \} 
        $

    \end{enumerate}
